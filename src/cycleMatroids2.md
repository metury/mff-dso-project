	8(" ");
	9(" ");

	1 -- "3.14" --- 2;

	4 -- "1.0" --- 0;
	0 -- "2.0" --- 5;
	1 -- "7.89" --- 6;
	2 -- "16.0" --- 7;
	3 -- "3.33" --- 8;
	4 -- "5.27" --- 9;
	5 -- "8.99" --- 7;
	5 -- "9.0" --- 8;
	6 -- "42.01" --- 8;
	6 -- "32.33" --- 9;
